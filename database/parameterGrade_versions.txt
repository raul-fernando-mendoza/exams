create or replace view thoth-dev-346022.thoth.parameter_grade_score_version as(
with a as (
select eg.applicationDay,o.organization_name, m.materia_name, me.exam_label, pg.label parameter_grade_label, avg(pg.score) as parameter_grade_score
from thoth-dev-346022.thoth.examGrades eg
join thoth-dev-346022.thoth.organizations o on eg.organization_id = o.id
join thoth-dev-346022.thoth.parameterGrades pg on pg.exam_grade_id = eg.id 
join thoth-dev-346022.thoth.materias m on m.id = eg.materia_id
join thoth-dev-346022.thoth.materias_exams me on me.id = eg.exam_id 
where eg.isDeleted=false and eg.isReleased = true and pg.version = 0  --and eg.id ='3a6715fe-60a7-48f3-8237-766b6f1df7d8'
group by 1,2,3,4,5
order by  1,2,3,4,5
)
,b as (
select eg.applicationDay,o.organization_name, m.materia_name, me.exam_label, pg.label parameter_grade_label, avg(pg.score) as parameter_grade_score_last
from thoth-dev-346022.thoth.examGrades eg
join thoth-dev-346022.thoth.organizations o on eg.organization_id = o.id
join thoth-dev-346022.thoth.parameterGrades pg on pg.exam_grade_id = eg.id 
join thoth-dev-346022.thoth.materias m on m.id = eg.materia_id
join thoth-dev-346022.thoth.materias_exams me on me.id = eg.exam_id 
where eg.isDeleted=false and eg.isReleased = true and pg.isCurrentVersion = true  --and eg.id ='3a6715fe-60a7-48f3-8237-766b6f1df7d8'
group by 1,2,3,4,5
order by  1,2,3,4,5
)
select a.*, b.parameter_grade_score_last
from a 
join b using( applicationDay, organization_name, materia_name, exam_label, parameter_grade_label ) 
)


select * from thoth-dev-346022.thoth.parameter_grade_score_version


create or replace view thoth-dev-346022.thoth.criteria_grade_score_version as(
with a as (
select eg.applicationDay,o.organization_name, m.materia_name, me.exam_label, pg.label parameter_grade_label, cg.label criteria_label, avg(cg.score * 10) score 
from thoth-dev-346022.thoth.examGrades eg
join thoth-dev-346022.thoth.organizations o on eg.organization_id = o.id
join thoth-dev-346022.thoth.parameterGrades pg on pg.exam_grade_id = eg.id 
join thoth-dev-346022.thoth.materias m on m.id = eg.materia_id
join thoth-dev-346022.thoth.materias_exams me on me.id = eg.exam_id 
join thoth-dev-346022.thoth.criteriaGrades cg on cg.parameter_grade_id = pg.id  
where eg.isDeleted=false and eg.isReleased = true and pg.version = 0  
and eg.id ='3a6715fe-60a7-48f3-8237-766b6f1df7d8'
group by 1,2,3,4,5,6
order by  1,2,3,4,5,6
)
,b as(
select eg.applicationDay,o.organization_name, m.materia_name, me.exam_label, pg.label parameter_grade_label, cg.label criteria_label, avg(cg.score * 10) score_latest
from thoth-dev-346022.thoth.examGrades eg
join thoth-dev-346022.thoth.organizations o on eg.organization_id = o.id
join thoth-dev-346022.thoth.parameterGrades pg on pg.exam_grade_id = eg.id 
join thoth-dev-346022.thoth.materias m on m.id = eg.materia_id
join thoth-dev-346022.thoth.materias_exams me on me.id = eg.exam_id 
join thoth-dev-346022.thoth.criteriaGrades cg on cg.parameter_grade_id = pg.id  
where eg.isDeleted=false and eg.isReleased = true and pg.isCurrentVersion = true
and eg.id ='3a6715fe-60a7-48f3-8237-766b6f1df7d8'
group by 1,2,3,4,5,6
order by  1,2,3,4,5,6
)
select a.*, b.score_latest
from a 
join b using ( applicationDay, organization_name, materia_name, exam_label, parameter_grade_label, criteria_label) 
)
